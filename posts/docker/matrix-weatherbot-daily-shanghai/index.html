Matrix Synapse 机器人：天气推送与服务器监控提醒

记录在自建 Matrix Synapse 上创建 bot 机器人账号，用 Open-Meteo 获取中国上海天气，并通过 cron 每天 05:00 自动推送给指定用户的完整部署流程。

这篇记录一个很实用的小自动化：在自建 Matrix Synapse 上创建一个真正的机器人账号 @bot:666666.xyz，每天早上 05:00 给指定用户 @honey:666666.xyz 推送中国上海天气预报；同时每 5 分钟检查磁盘空间、Docker 容器、TURN / LiveKit 视频通话相关服务，只有异常或恢复时才发提醒。

这个方案不是 Synapse Server Notices，而是普通 Matrix 机器人账号。好处是体验更像一个联系人：用户可以手动邀请它进私聊或房间，后续也可以把更多用户加入固定推送列表。

二、前置条件

本文假设 Synapse 在本机 8008 提供客户端 API：

http://127.0.0.1:8008

并且有一个可用的 Synapse 管理员 access token：

/root/.config/synapse-admin/access_token

先在服务器上确认 Synapse 正常：

curl -sS -o /tmp/synapse_versions.json -w 'http_code=%{http_code}\n' \
  http://127.0.0.1:8008/_matrix/client/versions
rm -f /tmp/synapse_versions.json

返回 http_code=200 就可以继续。

再准备脚本目录：

mkdir -p /root/data/docker_data/synapse/scripts
mkdir -p /root/.config/matrix-bot

import os
import secrets
import subprocess
import urllib.error
import urllib.parse
import urllib.request

BASE = "http://127.0.0.1:8008"
SERVER_NAME = "666666.xyz"
BOT_LOCALPART = "bot"
BOT_MXID = f"@{BOT_LOCALPART}:{SERVER_NAME}"
TARGET_MXID = "@honey:666666.xyz"
TOKEN_PATH = "/root/.config/synapse-admin/access_token"
BOT_TOKEN_PATH = "/root/.config/matrix-bot/access_token"
BOT_PASSWORD_PATH = "/root/.config/matrix-bot/bot-password"
SCRIPT_PATH = "/root/data/docker_data/synapse/scripts/matrix-bot-weather.py"
CONFIG_PATH = "/root/data/docker_data/synapse/scripts/matrix-bot-weather.json"

def req(method, url, token=None, data=None):
    headers = {}
    body = None
    if token:
        headers["Authorization"] = f"Bearer {token}"
    if data is not None:
        body = json.dumps(data, ensure_ascii=False).encode("utf-8")
        headers["Content-Type"] = "application/json"
    request = urllib.request.Request(url, data=body, headers=headers, method=method)
    try:
        with urllib.request.urlopen(request, timeout=20) as resp:
            raw = resp.read().decode("utf-8")
            return resp.status, json.loads(raw) if raw else {}
    except urllib.error.HTTPError as e:
        raw = e.read().decode("utf-8", errors="replace")
        try:
            payload = json.loads(raw)
        except Exception:
            payload = {"raw": raw}
        return e.code, payload

with open(TOKEN_PATH, "r", encoding="utf-8") as f:
    admin_token = f.read().strip()

os.makedirs(os.path.dirname(BOT_TOKEN_PATH), mode=0o700, exist_ok=True)

if not os.path.exists(BOT_PASSWORD_PATH):

        f.write(secrets.token_urlsafe(36) + "\n")

with open(BOT_PASSWORD_PATH, "r", encoding="utf-8") as f:

        "PUT",

            "displayname": "机器人",
            "admin": False,
            "deactivated": False,
            "user_type": None,

    raise SystemExit(f"check bot failed: HTTP {status} {payload}")

    "POST",
    f"{BASE}/_matrix/client/v3/login",

        "type": "m.login.password",

    "server_name": SERVER_NAME,
    "bot_user_id": BOT_MXID,
    "token_path": BOT_TOKEN_PATH,

创建：

写入：

#!/usr/bin/env python3
import datetime as dt
import hashlib
import html
import json
import os
import sys
import time
import urllib.error
import urllib.parse
import urllib.request

CONFIG_PATH = "/root/data/docker_data/synapse/scripts/matrix-bot-weather.json"

WEATHER_CODES = {
    0: "晴",
    1: "大部晴朗",
    2: "局部多云",
    3: "阴",
    45: "雾",
    48: "雾凇",
    51: "小毛毛雨",
    53: "毛毛雨",
    55: "较强毛毛雨",
    61: "小雨",
    63: "中雨",
    65: "大雨",
    71: "小雪",
    73: "中雪",
    75: "大雪",
    80: "小阵雨",
    81: "阵雨",
    82: "强阵雨",
    95: "雷雨",
}

def load_json(path, default):
    try:
        with open(path, "r", encoding="utf-8") as f:
            return json.load(f)
    except FileNotFoundError:
        return default

def save_json(path, data, mode=0o600):
    tmp = path + ".tmp"
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
        f.write("\n")
    os.chmod(tmp, mode)
    os.replace(tmp, path)

def request(method, url, token=None, data=None):
    headers = {}
    body = None
    if token:
        headers["Authorization"] = f"Bearer {token}"
    if data is not None:
        body = json.dumps(data, ensure_ascii=False).encode("utf-8")
        headers["Content-Type"] = "application/json"
    req = urllib.request.Request(url, data=body, headers=headers, method=method)
    try:
        with urllib.request.urlopen(req, timeout=30) as resp:
            raw = resp.read().decode("utf-8")
            return resp.status, json.loads(raw) if raw else {}
    except urllib.error.HTTPError as e:
        raw = e.read().decode("utf-8", errors="replace")
        try:
            payload = json.loads(raw)
        except Exception:
            payload = {"raw": raw}
        return e.code, payload

def quote(value):
    return urllib.parse.quote(value, safe="")

def get_weather(config):
    params = urllib.parse.urlencode(
        {
            "latitude": config["latitude"],
            "longitude": config["longitude"],
            "timezone": config.get("timezone", "Asia/Shanghai"),
            "forecast_days": 3,
            "daily": "weather_code,temperature_2m_max,temperature_2m_min,precipitation_probability_max,precipitation_sum,wind_speed_10m_max",
        }
    )
    url = "https://api.open-meteo.com/v1/forecast?" + params
    with urllib.request.urlopen(url, timeout=30) as resp:
        data = json.loads(resp.read().decode("utf-8"))

    daily = data["daily"]
    lines = [f"{config['city']}天气预报", ""]
    today = dt.date.today().isoformat()

    for i, day in enumerate(daily["time"]):
        label = "今天" if day == today else ("明天" if i == 1 else "后天")
        code = daily["weather_code"][i]
        desc = WEATHER_CODES.get(code, f"天气代码 {code}")
        tmin = daily["temperature_2m_min"][i]
        tmax = daily["temperature_2m_max"][i]
        pop = daily["precipitation_probability_max"][i]
        rain = daily["precipitation_sum"][i]
        wind = daily["wind_speed_10m_max"][i]
        lines.append(
            f"{label} {day}: {desc}，{tmin:.0f}-{tmax:.0f}℃，"
            f"降水概率 {pop}% ，降水量 {rain:.1f} mm，最大风速 {wind:.0f} km/h"
        )

    lines.extend(["", "数据源：Open-Meteo。每天 05:00 自动发送。"])
    return "\n".join(lines)

def sync_and_join_invites(config, token):
    base = config["homeserver"]
    state = load_json(config["state_path"], {})
    params = {"timeout": 0}
    if state.get("since"):
        params["since"] = state["since"]
    url = f"{base}/_matrix/client/v3/sync?{urllib.parse.urlencode(params)}"
    status, payload = request("GET", url, token)
    if status != 200:
        print(f"sync failed: HTTP {status} {payload}", file=sys.stderr)
        return
    for room_id in payload.get("rooms", {}).get("invite", {}).keys():
        join_url = f"{base}/_matrix/client/v3/rooms/{quote(room_id)}/join"
        status, join_payload = request("POST", join_url, token, {})
        if status not in (200, 201):
            print(f"join invite failed {room_id}: HTTP {status} {join_payload}", file=sys.stderr)
    state["since"] = payload.get("next_batch", state.get("since"))
    save_json(config["state_path"], state)

def is_room_encrypted(config, token, room_id):
    url = f"{config['homeserver']}/_matrix/client/v3/rooms/{quote(room_id)}/state/m.room.encryption/"
    status, _ = request("GET", url, token)
    return status == 200

def create_or_get_dm(config, token, user_id):
    base = config["homeserver"]
    state = load_json(config["state_path"], {})
    dms = state.setdefault("direct_rooms", {})
    old_room = dms.get(user_id)

    if old_room and not is_room_encrypted(config, token, old_room):
        return old_room

    if old_room:
        request("POST", f"{base}/_matrix/client/v3/rooms/{quote(old_room)}/leave", token, {})
        dms.pop(user_id, None)

    status, payload = request(
        "POST",
        f"{base}/_matrix/client/v3/createRoom",
        token,
        {
            "preset": "trusted_private_chat",
            "is_direct": True,
            "invite": [user_id],
            "name": "上海天气预报",
            "topic": "每天 05:00 自动推送中国上海天气预报",
        },
    )

    if status not in (200, 201) or "room_id" not in payload:
        raise RuntimeError(f"create DM for {user_id} failed: HTTP {status} {payload}")

    dms[user_id] = payload["room_id"]
    save_json(config["state_path"], state)
    return payload["room_id"]

def joined_rooms(config, token):
    status, payload = request("GET", f"{config['homeserver']}/_matrix/client/v3/joined_rooms", token)
    if status != 200:
        print(f"joined_rooms failed: HTTP {status} {payload}", file=sys.stderr)
        return []
    return payload.get("joined_rooms", [])

def send_room(config, token, room_id, body):
    base = config["homeserver"]
    txn = "shanghai-weather-" + dt.datetime.now(dt.timezone.utc).strftime("%Y%m%dT%H%M%S")
    txn += "-" + hashlib.sha1(room_id.encode("utf-8")).hexdigest()[:10]
    html_body = "<br>".join(html.escape(line) for line in body.splitlines())
    content = {
        "msgtype": "m.text",
        "body": body,
        "format": "org.matrix.custom.html",
        "formatted_body": html_body,
    }
    url = f"{base}/_matrix/client/v3/rooms/{quote(room_id)}/send/m.room.message/{txn}"
    status, payload = request("PUT", url, token, content)
    if status not in (200, 201):
        raise RuntimeError(f"send to {room_id} failed: HTTP {status} {payload}")

def main():
    config = load_json(CONFIG_PATH, None)
    if not config:
        raise SystemExit(f"missing config: {CONFIG_PATH}")

    with open(config["token_path"], "r", encoding="utf-8") as f:
        token = f.read().strip()

    sync_and_join_invites(config, token)
    body = get_weather(config)

    rooms = set(config.get("target_rooms", []))
    for user_id in config.get("target_users", []):
        rooms.add(create_or_get_dm(config, token, user_id))

    if config.get("send_to_joined_rooms", True):
        for room_id in joined_rooms(config, token):
            if not is_room_encrypted(config, token, room_id):
                rooms.add(room_id)

    sent = 0
    for room_id in sorted(rooms):
        send_room(config, token, room_id, body)
        sent += 1
        time.sleep(0.2)

    print(f"sent weather forecast to {sent} room(s)")

if __name__ == "__main__":
    main()

七、配置每天 5 点自动发送

创建包装脚本：

vi /root/data/docker_data/synapse/scripts/run-matrix-bot-weather.sh

写入：

#!/bin/sh
LOCK=/tmp/matrix-bot-weather.lock
TODAY=$(date +%F)
NOW=$(date +%H%M)
STAMP=/tmp/matrix-bot-weather.${TODAY}.stamp

# 只有 05:00 的定时任务才受当天 stamp 保护。
# 手动测试不会写 stamp，避免挡住下一次正点推送。
if [ "$NOW" = "0500" ] && [ -e "$STAMP" ]; then
  exit 0
fi

if mkdir "$LOCK" 2>/dev/null; then
  trap 'rmdir "$LOCK"' EXIT
  if /root/data/docker_data/synapse/scripts/matrix-bot-weather.py >> /var/log/matrix-bot-weather.log 2>&1; then
    [ "$NOW" = "0500" ] && touch "$STAMP"
  fi
fi

八、增加服务器监控提醒

天气预报是固定每天发送，监控提醒更适合“有事才吱声”。这里增加一个独立脚本，每 5 分钟检查一次：

  • 磁盘空间预警：检查 / 和 /var/lib/docker，默认使用率 80% 警告、90% 严重，或者可用空间低于 10GiB 警告。
  • Docker 容器异常通知：检查关键容器是否存在、是否运行、健康检查是否 unhealthy，以及重启次数是否增加。
  • TURN / LiveKit 状态检测：检查 coturn、LiveKit、LiveKit JWT、Element Call Web 相关容器、端口和 HTTP 探针。

import urllib.parse
import urllib.request

CONFIG_PATH = "/root/data/docker_data/synapse/scripts/matrix-bot-monitor.json"

def load_json(path, default):
    try:
        with open(path, "r", encoding="utf-8") as f:
            return json.load(f)
    except FileNotFoundError:
        return default

def save_json(path, data):
    tmp = path + ".tmp"
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
        f.write("\n")
    os.chmod(tmp, 0o600)
    os.replace(tmp, path)

def run(cmd):
    return subprocess.run(cmd, text=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE, check=False)

def request(method, url, token=None, data=None, timeout=20):
    headers = {}
    body = None
    if token:
        headers["Authorization"] = f"Bearer {token}"
    if data is not None:
        body = json.dumps(data, ensure_ascii=False).encode("utf-8")
        headers["Content-Type"] = "application/json"
    req = urllib.request.Request(url, data=body, headers=headers, method=method)
    try:
        with urllib.request.urlopen(req, timeout=timeout) as resp:
            raw = resp.read().decode("utf-8", errors="replace")
            if not raw:
                return resp.status, {}
            try:
                return resp.status, json.loads(raw)
            except Exception:
                return resp.status, {"raw": raw[:200]}
    except Exception as e:
        return 0, {"error": str(e)}

def quote(value):
    return urllib.parse.quote(value, safe="")

def check_disks(config):
    issues = []
    disk = config["disk"]
    for mount in disk["mounts"]:
        proc = run(["df", "-P", "-B1", mount])
        if proc.returncode != 0:
            issues.append(f"[disk] {mount}: df failed")
            continue
        parts = proc.stdout.strip().splitlines()[-1].split()
        used_percent = int(parts[4].rstrip("%"))
        free_gb = int(parts[3]) / 1024 / 1024 / 1024
        if used_percent >= disk["critical_percent"]:
            issues.append(f"[disk:critical] {parts[5]} used={used_percent}% free={free_gb:.1f}GiB")
        elif used_percent >= disk["warn_percent"] or free_gb < disk["min_free_gb"]:
            issues.append(f"[disk:warning] {parts[5]} used={used_percent}% free={free_gb:.1f}GiB")
    return issues

def docker_inspect(names):
    result = {}
    for name in names:
        proc = run(["docker", "inspect", "--format", "{{json .}}", name])
        if proc.returncode != 0:
            result[name] = {"missing": True}
            continue
        data = json.loads(proc.stdout)
        state = data.get("State", {})
        health = state.get("Health") or {}
        result[name] = {
            "missing": False,
            "running": bool(state.get("Running")),
            "status": state.get("Status", "unknown"),
            "health": health.get("Status", "none"),
            "restart_count": int(data.get("RestartCount", 0)),
        }
    return result

def check_docker(config, state):
    issues = []
    previous = state.get("docker_restart_counts", {})
    current = {}
    for name, info in docker_inspect(config["docker"]["expected_containers"]).items():
        if info.get("missing"):
            issues.append(f"[docker] {name}: missing")
            continue
        current[name] = info["restart_count"]
        if not info["running"]:
            issues.append(f"[docker] {name}: not running, status={info['status']}")
        if info["health"] == "unhealthy":
            issues.append(f"[docker] {name}: health=unhealthy")
        old = previous.get(name)
        if old is not None and info["restart_count"] > old:
            issues.append(f"[docker] {name}: restart count {old} -> {info['restart_count']}")
    state["docker_restart_counts"] = current
    return issues

def tcp_open(port):
    try:
        with socket.create_connection(("127.0.0.1", int(port)), timeout=3):
            return True
    except Exception:
        return False

def udp_listening(port):
    proc = run(["ss", "-lun"])
    return proc.returncode == 0 and any(f":{port}" in line for line in proc.stdout.splitlines())

def check_realtime(config):
    issues = []
    rt = config["realtime"]
    for name, info in docker_inspect(rt["containers"]).items():
        if info.get("missing"):
            issues.append(f"[realtime] {name}: missing")
        elif not info["running"]:
            issues.append(f"[realtime] {name}: not running, status={info['status']}")
    for port in rt["tcp_ports"]:
        if not tcp_open(port):
            issues.append(f"[realtime] tcp/{port}: not listening")
    for port in rt["udp_ports"]:
        if not udp_listening(port):
            issues.append(f"[realtime] udp/{port}: not listening")
    for probe in rt["http_probes"]:
        status, payload = request("GET", probe["url"], timeout=5)
        if status not in probe["ok"]:
            issues.append(f"[realtime] {probe['name']}: HTTP {status} {payload}")
    return issues

def send_room(config, token, room_id, body):

    txn += "-" + hashlib.sha1((room_id + body).encode("utf-8")).hexdigest()[:10]
    html_body = "<br>".join(html.escape(line) for line in body.splitlines())
    content = {
        "msgtype": "m.text",
        "body": body,
        "format": "org.matrix.custom.html",
        "formatted_body": html_body,
    }
    url = f"{config['homeserver']}/_matrix/client/v3/rooms/{quote(room_id)}/send/m.room.message/{txn}"
    status, payload = request("PUT", url, token, content)
    if status not in (200, 201):
        raise RuntimeError(f"send failed: HTTP {status} {payload}")

def send_alert(config, token, state, body):
    rooms = set(config.get("target_rooms", []))
    weather_state = load_json(config.get("bot_weather_state_path", ""), {})
    for user_id, room_id in weather_state.get("direct_rooms", {}).items():
        if user_id in config["target_users"]:
            rooms.add(room_id)
    for room_id in sorted(rooms):
        send_room(config, token, room_id, body)
        time.sleep(0.2)

def main():
    config = load_json(CONFIG_PATH, None)
    if not config:
        raise SystemExit(f"missing config: {CONFIG_PATH}")
    state = load_json(config["state_path"], {})
    with open(config["token_path"], "r", encoding="utf-8") as f:
        token = f.read().strip()

    issues = sorted(set(check_disks(config) + check_docker(config, state) + check_realtime(config)))
    previous = set(state.get("active_issues", []))
    current = set(issues)
    new_issues = sorted(current - previous)
    recovered = sorted(previous - current)

    state["active_issues"] = issues
    state["last_check"] = dt.datetime.now().isoformat(timespec="seconds")

    if new_issues or recovered:
        lines = [f"服务器监控提醒 {dt.datetime.now().strftime('%Y-%m-%d %H:%M:%S')}", ""]
        if new_issues:
            lines.append("新增异常：")
            lines.extend(f"- {x}" for x in new_issues)
        if recovered:
            if new_issues:
                lines.append("")
            lines.append("已恢复：")
            lines.extend(f"- {x}" for x in recovered)
        send_alert(config, token, state, "\n".join(lines))

    save_json(config["state_path"], state)
    print(f"monitor ok: active={len(issues)} new={len(new_issues)} recovered={len(recovered)}")

if __name__ == "__main__":
    main()

十、给其他用户添加推送

有两种方式。

第一种，用户直接在 Matrix 客户端里邀请：

机器人每天运行时会先同步邀请并自动加入。只要它加入的是未加密房间，后续每天 05:00 都会发天气。

]

十一、关于加密房间

这个脚本使用的是普通 Matrix Client-Server API，没有实现端到端加密设备管理，也没有处理 Olm/Megolm 密钥。

因此建议让机器人使用未加密私聊或未加密房间。脚本里也会跳过已经启用 m.room.encryption 的房间，避免在加密房间里发出客户端无法正常处理的明文消息。

如果一定要在端到端加密房间里使用机器人，需要换成支持 E2EE 的机器人框架，例如 matrix-nio[e2e]，并额外处理设备信任、密钥存储和恢复。
